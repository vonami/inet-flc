module FailedRouter
    parameters:
        namid: numeric,
        routingFile: string;
    gates:
        in: in[];
        out: out[];
        in: ethIn[];
        out: ethOut[];
    submodules:
        dummy: Dummy;
            parameters:
                namid = namid;
            gatesizes:
                in[sizeof(in)],
                out[sizeof(out)],
                ethIn[sizeof(ethIn)],
                ethOut[sizeof(ethOut)];
    connections nocheck:
        for i=0..sizeof(out)-1 do
            in[i] --> dummy.in[i];
            dummy.out[i] --> out[i];
        endfor;
        for i=0..sizeof(ethOut)-1 do
            ethIn[i] --> dummy.ethIn[i];
            dummy.ethOut[i] --> ethOut[i];
        endfor;
endmodule
