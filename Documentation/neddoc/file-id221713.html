//
// Accepts any number of incoming TCP connections, and discards whatever
// arrives on them.
// Compatible with both IPv4 and IPv6.
//
simple TCPSinkApp
    parameters:
        address: string, // may be left empty ("")
        port: numeric const; // port number to listen on
    gates:
        in: tcpIn;
        out: tcpOut;
endsimple
